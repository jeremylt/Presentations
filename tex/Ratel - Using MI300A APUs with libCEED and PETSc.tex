MatCeedSetTime(Mat mat, PetscReal time);

// COO assembly support
MatCeedCreateMatCOO(Mat mat_ceed, Mat *mat_coo);
MatCeedSetPreallocationCOO(Mat mat_ceed, Mat mat_coo);

MatMultTranspose_Ceed(Mat A, Vec Y, Vec X)

MatInvertBlockDiagonal_Ceed(Mat mat_ceed, const PetscScalar **values);

\begin{frame}

\frametitle{Porting Outcomes}

\begin{itemize}

\item PETSc and libCEED changes to support APUs\\

~\\

\begin{itemize}

\item Allocate all memory as device memory\\

~\\

\item Treat same pointer as host memory\\

~\\

\item Must forbid mismatched memory operations (\lstinline{memset})\\

\end{itemize}

~\\

\item Unified memory improves performance on APU hardware\\

~\\

\begin{itemize}

\item At least 5\% speedup with unified memory\\

\end{itemize}

\end{itemize}

\end{center}
